75   class FloatSpinbox(customtkinter.CTkFrame):
76       def __init__(self, *args, width: int = 100, height: int = 32,
77                    step_size: Union[int, float] = 1, command: Callable = None, **kwargs):
78           super().__init__(*args, width=width, height=height, **kwargs)
79   
80           self.step_size = step_size
81           self.command = command
82   
83           self.configure(fg_color=("gray78", "gray28"))
84           self.grid_columnconfigure((0, 2), weight=0)
85           self.grid_columnconfigure(1, weight=1)
86   
87           self.subtract_button = customtkinter.CTkButton(
88               self, text="-", width=height - 6, height=height - 6, command=self.subtract_button_callback)
89           self.subtract_button.grid(row=0, column=0, padx=(3, 0), pady=3)
90   
91           self.entry = customtkinter.CTkEntry(self, width=width - (2 * height), height=height - 6, border_width=0)
92           self.entry.grid(row=0, column=1, columnspan=1, padx=3, pady=3, sticky="ew")
93   
94           self.add_button = customtkinter.CTkButton(
95               self, text="+", width=height - 6, height=height - 6, command=self.add_button_callback)
96           self.add_button.grid(row=0, column=2, padx=(0, 3), pady=3)
97   
98           self.entry.insert(0, "0.0")
99   
100      def add_button_callback(self):
101          if self.command is not None:
102              self.command()
103          try:
104              value = float(self.entry.get()) + self.step_size
105              self.entry.delete(0, "end")
106              self.entry.insert(0, "{:.2f}".format(value))
107          except ValueError:
108              return
109  
110      def subtract_button_callback(self):
111          if self.command is not None:
112              self.command()
113          try:
114              value = float(self.entry.get()) - self.step_size
115              self.entry.delete(0, "end")
116              self.entry.insert(0, "{:.2f}".format(value))
117          except ValueError:
118              return
119  
120      def get(self) -> Union[float, None]:
121          try:
122              return float(self.entry.get())
123          except ValueError:
124              return None
125  
126      def set(self, value: float):
127          self.entry.delete(0, "end")
128          self.entry.insert(0, "{:.2f}".format(value))

225  class DimensionWindow(ctk.CTkFrame):
226      def __init__(self, master, df):
227          super().__init__(master)
228  
229          self.df = df
230  
231          self.grid_rowconfigure(0, weight=1)
232          self.grid_columnconfigure(0, weight=1)
233  
234          self.max_dimension_x = 1200
235          self.max_dimension_y = 1500
236  
237          self.fig, self.ax = plt.subplots(figsize=(5, 5))
238          self.ax.set_xlim(0, self.max_dimension_x)
239          self.ax.set_ylim(0, self.max_dimension_y)
240          self.ax.set_facecolor('lightslategray')
241          self.ax.set_aspect('equal')
242  
243          self.rect = plt.Rectangle((0, 0), self.max_dimension_x / 2, self.max_dimension_y / 2, fc='lightsteelblue')
244          self.ax.add_patch(self.rect)
245  
246          self.canvas = FigureCanvasTkAgg(self.fig, master=self)
247          self.canvas.get_tk_widget().grid(row=0, column=0, sticky="nsew")
248  
249          # Sliders
250          self.width_slider = tk.Scale(self, from_=0, to=self.max_dimension_x, length=400,
251                                       command=self.update_rectangle, orient='horizontal')
252          self.width_slider.set(self.max_dimension_x / 2)  # Set initial value to half of the maximum
253          self.width_slider.grid(row=1, column=0, sticky="ew")
254  
255          self.height_slider = tk.Scale(self, from_=self.max_dimension_y, to=0, length=400,
256                                        command=self.update_rectangle, orient='vertical')
257          self.height_slider.set(self.max_dimension_y / 2)  # Set initial value to half of the maximum
258          self.height_slider.grid(row=0, column=1, sticky="ns")
259  
260      def update_axes_limits(self):
261          self.ax.set_xlim(0, self.max_dimension_x)
262          self.ax.set_ylim(0, self.max_dimension_y)
263  
264      def update_rectangle(self, _):
265          width = self.width_slider.get()
266          height = self.height_slider.get()
267  
268          if width > height:
269              width = height
270              self.width_slider.set(width)
271  
272          self.rect.set_width(width)
273          self.rect.set_height(height)
274          self.fig.canvas.draw_idle()
275  
276          alpha, beta = self.calculate_ratio_and_get_alpha_beta(height, width)
277  
278      def calculate_ratio_and_get_alpha_beta(self, length, width):
279          ratio = length / width if width != 0 else float('inf')
280  
281          # Find the closest column
282          closest_column = min(self.df.columns,
283                               key=lambda x: abs(x - ratio) if x != 'infinity' else abs(float('inf') - ratio))
284  
285          # Get alpha and beta
286          alpha = self.df.loc['alpha', closest_column]
287          beta = self.df.loc['beta', closest_column]
288  
289          return alpha, beta

628  class ExternalLoadLending:
629      """Class to compute and return the external load lending."""
630  
631      def __init__(self, side_length_short, side_length_long, gap_width, outer_thickness, inner_thickness, external_load):
632          """Initialize the class with side lengths, gap width, thicknesses, and external load."""
633  
634          self.side_length_short = side_length_short
635          self.side_length_long = side_length_long
636          self.gap_width = gap_width
637          self.outer_thickness = outer_thickness
638          self.inner_thickness = inner_thickness
639          self.external_load = external_load
640  
641      def compute_z1(self):
642          """Compute and return the value of z1 based on the side lengths."""
643  
644          _exp = (-1.123 * ((self.side_length_long / self.side_length_short) - 1) ** 1.097)
645          return 181.8 * (self.side_length_short / self.side_length_long) ** 2 * (
646                  0.00406 + 0.00896 * (1 - math.exp(_exp)))
647  
648      def compute_chi(self):
649          """Compute and return the value of chi based on the side lengths and z1."""
650  
651          z1 = self.compute_z1()
652          _exp = (-6.8 * (self.side_length_long / self.side_length_short) ** 1.33)
653          return (z1 / 16) * (0.4198 + 0.22 * math.exp(_exp)) * (self.side_length_long / self.side_length_short) ** 2
654  
655      def compute_a_(self):
656          """Compute and return the value of a_ based on the thicknesses, gap width, and chi."""
657  
658          chi = self.compute_chi()
659          return 29.9 * ((self.gap_width * (self.outer_thickness ** 3) * (self.inner_thickness ** 3)) / (
660                  ((self.outer_thickness ** 3) + (self.inner_thickness ** 3)) * chi)) ** 0.25
661  
662      def compute_isolation_factor(self):
663          """Compute and return the isolation factor based on the side lengths and a_."""
664  
665          a_ = self.compute_a_()
666          return 1 / (1 + (self.side_length_short / a_) ** 4)
667  
668      def compute_pe(self):
669          """Compute and return the value of P_E based on the isolation factor, thicknesses, and external load."""
670  
671          isolation_factor = self.compute_isolation_factor()
672          return (1 - isolation_factor) * ((self.inner_thickness ** 3) / (
673                  (self.outer_thickness ** 3) + (self.inner_thickness ** 3))) * self.external_load
